Core Calculations

function reduceNumber(number, preserveMaster = true) {
  const masterNumbers = [11, 22, 33];
  while (number > 9) {
    if (preserveMaster && masterNumbers.includes(number)) return number;
    number = String(number).split("").reduce((sum, digit) => sum + Number(digit), 0);
  }
  return number;
}

function calculateLifePath(dateString) {
  // dateString format: MM/DD/YYYY
  const digits = dateString.replace(/\D/g, "");
  const total = digits.split("").reduce((sum, digit) => sum + Number(digit), 0);
  return { raw: total, reduced: reduceNumber(total, true) };
}

function calculateBirthDay(dateString) {
  const day = Number(dateString.split("/")[1]);
  return { raw: day, reduced: reduceNumber(day, true) };
}

function letterValue(letter) {
  const map = {
    A:1,J:1,S:1, B:2,K:2,T:2, C:3,L:3,U:3,
    D:4,M:4,V:4, E:5,N:5,W:5, F:6,O:6,X:6,
    G:7,P:7,Y:7, H:8,Q:8,Z:8, I:9,R:9
  };
  return map[letter.toUpperCase()] || 0;
}

function calculateExpression(fullName) {
  const clean = fullName.replace(/[^A-Za-z]/g, "");
  const total = clean.split("").reduce((sum, char) => sum + letterValue(char), 0);
  return { raw: total, reduced: reduceNumber(total, true) };
}

function calculateSoulUrge(fullName) {
  const vowels = fullName.replace(/[^A-Za-z]/g, "").split("").filter(char => "AEIOUaeiou".includes(char));
  const total = vowels.reduce((sum, char) => sum + letterValue(char), 0);
  return { raw: total, reduced: reduceNumber(total, true) };
}

function calculatePersonality(fullName) {
  const consonants = fullName.replace(/[^A-Za-z]/g, "").split("").filter(char => !"AEIOUaeiou".includes(char));
  const total = consonants.reduce((sum, char) => sum + letterValue(char), 0);
  return { raw: total, reduced: reduceNumber(total, true) };
}

function calculateMaturity(lifePathReduced, expressionReduced) {
  const total = lifePathReduced + expressionReduced;
  return { raw: total, reduced: reduceNumber(total, true) };
}
